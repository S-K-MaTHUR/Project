package pack03;


import java.io.*;
import java.util.*;

public class passwords {

    public static String generateRandomPassword(int length){
if(length<8){
    return "Invalid Password!";
}

    String specialCharacters = "!@#$%^&*";
    String combineLetters = "ABCDEFGHIJKLMNOPQRSTUVWXYZ1234567890abcdefghijklmnopqrstuvwxyz!@#$%^&*";

    String password = "";
    Random random = new Random();

    password+=(char)(random.nextInt(26)+'A');
    password+=(char)(random.nextInt(26)+'a');
    password+=random.nextInt(10);
    password+=specialCharacters.charAt(random.nextInt(specialCharacters.length()));

    for(int i=4;i<=length;i++){
        password+=combineLetters.charAt(random.nextInt(combineLetters.length()));
    }

    return password;
}
    
    public static void main(String[] args) {
        
        System.out.println(generateRandomPassword(5));
    }
}
